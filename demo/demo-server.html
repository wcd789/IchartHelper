 22     /**
 23      * 
 24      * @return

 26     public String getDomoCurve() {
 27         try {
 28 
 29             List<Object> valuesSH = this.getValuesSH();//模拟生成上海的数据（实际应用可能来时数据库）
 30             List<Object> valuesBJ = this.getValuesBJ();
 31             List<String> labels = this.getLabels();//模拟生成labels标签
 32 
 33             List<IchartNVsCL> nVsCLs = new ArrayList<IchartNVsCL>();
 34             nVsCLs.add(new IchartNVsCL("上海", valuesSH, "#aad0db", 2));
 35             nVsCLs.add(new IchartNVsCL("北京", valuesBJ, "#f68f70", 2));
 36             //new IchartNVsCL(...)有几个重载方法,区别在于如果不指定line_width默认为2
 37             //不指定color将会随机生成一个颜色
 38             //1.public IchartNVsCL(String name, List<Object> values, String color, int line_width) 
 39             //2.public IchartNVsCL(String name, List<Object> values, String color)
 40             //3. public IchartNVsCL(String name, List<Object> values)
 41 
 42             //以下两行代码会生成json代码,如下（以下部分icharhelp帮你完成，实际上就这两句话，简单吧）
 43             Area2D area2D = new Area2D(nVsCLs, labels);
 44             return area2D.toIchartJson();
 45         } catch (IOException ex) {
 46             return "";
 47         }
 48     }

 67     /**
 68      * 模拟生成labels 主要生成页面数据 var labels =
 69      * ["01月","02月","03月","04月","05月","06月","07月","08月","09月","10月","11月","12月"];

 73     private List<String> getLabels() {
 74         List<String> lables = new ArrayList<String>();
 75         String[] labelstr = {"01月", "02月", "03月", "04月", "05月", "06月", "07月", "08月", "09月", "10月", "11月", "12月"};
 76         lables.addAll(Arrays.asList(labelstr));
 77         return lables;
 78     }

 80     /**
 81      * 模拟生成上海数据 主要生成页面 [4,16,18,24,32,36,38,38,36,26,20,14]
 82      *
 83      * @return
 84      */
 85     private List<Object> getValuesSH() {
 86         List<Object> valuesSH = new ArrayList<Object>();
 87         int[] values = {4, 16, 18, 24, 32, 36, 38, 38, 36, 26, 20, 14};
 88         valuesSH.addAll(Arrays.asList(values));
 89         return valuesSH;
 90     }
